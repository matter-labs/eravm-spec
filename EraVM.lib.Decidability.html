Library EraVM.lib.Decidability

From Coq.Logic Require Decidable.

Import Decidable.

Section Decidability.

  Definition dec(A: Prop) := { A } + { ¬ A }.

  Lemma dec_decidable :
    ∀ P, dec P → decidable P.

  Theorem dec_conj:
    ∀ A B: Prop,
      dec A →
      dec B →
      dec (A ∧ B).

  Theorem dec_not : ∀ A:Prop, dec A → dec (¬ A).

  Theorem dec_not_not : ∀ P:Prop, dec P → ¬ ¬ P → P.

  Theorem not_not : ∀ P: Prop,
      dec P → ¬ ¬ P → P.

  Theorem dec_or:
    ∀ A B: Prop,
      dec A →
      dec B →
      dec (A ∨ B).

  Definition eq_dec (T:Type) := ∀ x1 x2: T, dec (x1 = x2).

End Decidability.

Ltac decide_equality :=
  unfold eq_dec, dec in *;
  repeat decide equality;
  unfold eq_dec, dec in *;
  eauto with decidable_prop.
